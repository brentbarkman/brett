```typescript
import type { MeetingTranscriptTurn, MeetingNoteAttendee } from "@brett/types";

/** Summary priority: granola > google_meet (Granola produces richer AI summaries) */
const SUMMARY_PRIORITY: Record<string, number> = { granola: 10, google_meet: 5 };

export interface MergeInput {
  title: string | null;
  summary: string | null;
  transcript: MeetingTranscriptTurn[] | null;
  attendees: MeetingNoteAttendee[] | null;
  sources: string[];
}

export interface SourceInput {
  provider: string;
  title: string;
  summary: string | null;
  transcript: MeetingTranscriptTurn[] | null;
  attendees: MeetingNoteAttendee[] | null;
}

export interface MergeResult {
  title: string;
  summary: string | null;
  transcript: MeetingTranscriptTurn[] | null;
  attendees: MeetingNoteAttendee[] | null;
  sources: string[];
}

export function mergeMeetingNoteFields(existing: MergeInput, source: SourceInput): MergeResult {
  const title = existing.title ?? source.title;

  // Summary: prefer higher-priority provider
  let summary = existing.summary;
  if (source.summary) {
    if (!existing.summary) {
      summary = source.summary;
    } else {
      const existingPriority = Math.max(...existing.sources.map((s) => SUMMARY_PRIORITY[s] ?? 0));
      const sourcePriority = SUMMARY_PRIORITY[source.provider] ?? 0;
      if (sourcePriority > existingPriority) {
        summary = source.summary;
      }
    }
  }

  // Transcript: prefer longer (more speaker turns = richer)
  let transcript = existing.transcript;
  if (source.transcript && source.transcript.length > 0) {
    if (!existing.transcript || source.transcript.length > existing.transcript.length) {
      transcript = source.transcript;
    }
  }

  // Attendees: union by email (case-insensitive)
  const attendees = mergeAttendees(existing.attendees, source.attendees);

  // Sources: append if new
  const sources = existing.sources.includes(source.provider)
    ? existing.sources
    : [...existing.sources, source.provider];

  return { title, summary, transcript, attendees, sources };
}

function mergeAttendees(
  existing: MeetingNoteAttendee[] | null,
  incoming: MeetingNoteAttendee[] | null,
): MeetingNoteAttendee[] | null {
  if (!existing && !incoming) return null;
  if (!existing) return incoming;
  if (!incoming) return existing;

  const seen = new Map<string, MeetingNoteAttendee>();
  for (const a of existing) seen.set(a.email.toLowerCase(), a);
  for (const a of incoming) {
    if (!seen.has(a.email.toLowerCase())) seen.set(a.email.toLowerCase(), a);
  }
  return Array.from(seen.values());
}
```

```typescript
/**
 * Scrub PII from provider raw data before storage.
 * Follows the same pattern as scrubRawEvent() in calendar-sync.ts.
 */
export function scrubProviderRawData(provider: string, rawData: unknown): unknown {
  if (!rawData || typeof rawData !== "object") return rawData;

  switch (provider) {
    case "granola":
      return scrubGranolaRawData(rawData as Record<string, unknown>);
    case "google_meet":
      return scrubGoogleMeetRawData(rawData as Record<string, unknown>);
    default:
      return rawData;
  }
}

function scrubGranolaRawData(data: Record<string, unknown>): Record<string, unknown> {
  const { attendees, participants, known_participants, ...safe } = data;
  return {
    ...safe,
    // Strip emails, keep count for debugging
    attendeeCount: Array.isArray(attendees) ? attendees.length : undefined,
  };
}

function scrubGoogleMeetRawData(data: Record<string, unknown>): Record<string, unknown> {
  const { suggestionsViewMode, namedStyles, suggestedInsertions, suggestedDeletions, ...safe } = data;
  return safe;
}
```

```typescript
import type { MeetingNoteProvider } from "./types.js";

class MeetingProviderRegistry {
  private providers: MeetingNoteProvider[] = [];

  register(provider: MeetingNoteProvider): void {
    this.providers.push(provider);
  }

  async getAvailable(userId: string): Promise<MeetingNoteProvider[]> {
    const results = await Promise.all(
      this.providers.map(async (p) => ({ provider: p, available: await p.isAvailable(userId) })),
    );
    return results.filter((r) => r.available).map((r) => r.provider);
  }

  getAll(): MeetingNoteProvider[] {
    return [...this.providers];
  }
}

export const providerRegistry = new MeetingProviderRegistry();
```

```typescript
import type { CalendarEvent } from "@prisma/client";
import type { MeetingNoteProvider, ProviderMeetingData } from "./types.js";
import { prisma } from "../../lib/prisma.js";
import { withGranolaClient, type GranolaMeetingListItem } from "../../lib/granola-mcp.js";
import { findBestMatch, type MatchCandidate } from "../meeting-matcher.js";
import type { MeetingTranscriptTurn, MeetingNoteAttendee } from "@brett/types";

export class GranolaProvider implements MeetingNoteProvider {
  readonly provider = "granola";

  async isAvailable(userId: string): Promise<boolean> {
    const account = await prisma.granolaAccount.findUnique({
      where: { userId },
      select: { id: true, tokenExpiresAt: true },
    });
    return !!account;
  }

  async fetchForEvent(userId: string, calendarEvent: CalendarEvent): Promise<ProviderMeetingData | null> {
    const account = await prisma.granolaAccount.findUnique({ where: { userId } });
    if (!account) return null;

    // Use 15min-before to 30min-after window (existing logic from granola-sync.ts)
    const windowStart = new Date(calendarEvent.startTime.getTime() - 15 * 60 * 1000);
    const windowEnd = new Date(calendarEvent.endTime.getTime() + 30 * 60 * 1000);

    return await withGranolaClient(account.id, async (tools) => {
      const meetings = await tools.listMeetings("custom", windowStart, windowEnd);
      if (meetings.length === 0) return null;

      // Match to this specific calendar event
      const candidates: MatchCandidate[] = [{
        id: calendarEvent.id,
        title: calendarEvent.title,
        startTime: calendarEvent.startTime,
        endTime: calendarEvent.endTime,
        attendees: (calendarEvent.attendees as MeetingNoteAttendee[] | null) ?? [],
      }];

      for (const meeting of meetings) {
        const match = findBestMatch(
          {
            title: meeting.title,
            startTime: new Date(meeting.startTime),
            endTime: new Date(meeting.endTime),
            attendees: meeting.attendees ?? [],
          },
          candidates,
        );

        if (match) {
          const details = await tools.getMeetings([meeting.id]);
          const detail = details[0];
          if (!detail) continue;

          const transcriptResult = await tools.getTranscript(meeting.id);

          return {
            provider: "granola",
            externalId: meeting.id,
            accountId: account.id,
            title: detail.title,
            summary: detail.summary ?? null,
            transcript: transcriptResult?.turns as MeetingTranscriptTurn[] | null ?? null,
            attendees: (meeting.attendees as MeetingNoteAttendee[] | null) ?? null,
            meetingStartedAt: new Date(meeting.startTime),
            meetingEndedAt: new Date(meeting.endTime),
            rawData: detail,
          };
        }
      }
      return null;
    });
  }

  async fetchRecent(userId: string, since: Date, until: Date): Promise<ProviderMeetingData[]> {
    const account = await prisma.granolaAccount.findUnique({ where: { userId } });
    if (!account) return [];

    return await withGranolaClient(account.id, async (tools) => {
      const meetings = await tools.listMeetings("custom", since, until);
      if (meetings.length === 0) return [];

      const results: ProviderMeetingData[] = [];
      const details = await tools.getMeetings(meetings.map((m) => m.id));

      for (let i = 0; i < meetings.length; i++) {
        const meeting = meetings[i]!;
        const detail = details[i];

        const transcriptResult = await tools.getTranscript(meeting.id);

        results.push({
          provider: "granola",
          externalId: meeting.id,
          accountId: account.id,
          title: detail?.title ?? meeting.title,
          summary: detail?.summary ?? null,
          transcript: transcriptResult?.turns as MeetingTranscriptTurn[] | null ?? null,
          attendees: (meeting.attendees as MeetingNoteAttendee[] | null) ?? null,
          meetingStartedAt: new Date(meeting.startTime),
          meetingEndedAt: new Date(meeting.endTime),
          rawData: detail ?? null,
        });
      }

      return results;
    });
  }
}
```

```typescript
/** Escape single quotes for Drive API q parameter */
export function escapeDriveQuery(input: string): string {

interface DocParagraph {
  type: string;
  text: string;

/**
 * Find Meet transcript and notes docs for a calendar event.
 * Strategy:
 *   1. Check event attachments for Google Docs
 *   2. Fallback: search Drive by event title + time window
 */
export async function findMeetArtifacts(
  driveClient: drive_v3.Drive,
  attachments: Array<{ fileId?: string; title?: string; mimeType?: string }> | null,
  eventTitle: string,
  eventStart: Date,
  eventEnd: Date,
): Promise<{ transcriptFileId: string | null; notesFileId: string | null }> {
  let transcriptFileId: string | null = null;
  let notesFileId: string | null = null;

  // 1. Check attachments first (reliable path)
  if (attachments) {
    for (const att of attachments) {
      if (att.mimeType !== "application/vnd.google-apps.document") continue;
      if (!att.fileId) continue;

      const title = (att.title ?? "").toLowerCase();
      if (title.startsWith("transcript")) {
        transcriptFileId = att.fileId;
      } else if (!notesFileId) {
        notesFileId = att.fileId;
      }
    }
  }

  // 2. Fallback: Drive search (only if attachments didn't find everything)
  if (!transcriptFileId || !notesFileId) {
    const searchStart = new Date(eventStart.getTime() - 60 * 60 * 1000); // 1h before
    const searchEnd = new Date(eventEnd.getTime() + 2 * 60 * 60 * 1000); // 2h after
    const escaped = escapeDriveQuery(eventTitle);

    try {
      const res = await driveClient.files.list({
        q: `mimeType='application/vnd.google-apps.document' and name contains '${escaped}' and createdTime > '${searchStart.toISOString()}' and createdTime < '${searchEnd.toISOString()}'`,
        fields: "files(id,name,createdTime)",
        pageSize: 10,
      });

      const files = res.data.files ?? [];
      // Tiebreaker: prefer doc created closest to event end
      files.sort((a, b) => {
        const aTime = a.createdTime ? Math.abs(new Date(a.createdTime).getTime() - eventEnd.getTime()) : Infinity;
        const bTime = b.createdTime ? Math.abs(new Date(b.createdTime).getTime() - eventEnd.getTime()) : Infinity;
        return aTime - bTime;
      });

      for (const file of files) {
        if (!file.id) continue;
        const name = (file.name ?? "").toLowerCase();
        if (!transcriptFileId && name.startsWith("transcript")) {
          transcriptFileId = file.id;
        } else if (!notesFileId) {
          notesFileId = file.id;
        }
      }
    } catch (err) {
      console.warn("[google-drive] Drive search failed:", err);
    }
  }

  return { transcriptFileId, notesFileId };
}

/**
 * Read a Google Doc and extract paragraphs.
 * Returns a flat list of { type, text } for each structural element.
 */
export async function readDocContent(docsClient: docs_v1.Docs, fileId: string): Promise<DocParagraph[]> {
  const doc = await docsClient.documents.get({ documentId: fileId });
  const content = doc.data.body?.content ?? [];
  const paragraphs: DocParagraph[] = [];

  for (const element of content) {
    if (element.paragraph) {
      const text = element.paragraph.elements
        ?.map((e) => e.textRun?.content ?? "")
        .join("")
        .replace(/\n$/, "") ?? "";
      paragraphs.push({ type: "paragraph", text });
    }
  }

  return paragraphs;
}

const TIMESTAMP_SPEAKER_RE = /^\[(\d{1,2}:\d{2}:\d{2})\]\s+(.+)$/;

/** Parse a Google Meet transcript doc into speaker turns */
export function parseTranscriptDoc(content: DocParagraph[]): MeetingTranscriptTurn[] {
  const turns: MeetingTranscriptTurn[] = [];
  let currentSpeaker: string | null = null;
  let currentLines: string[] = [];

  function flushTurn() {
    if (currentSpeaker && currentLines.length > 0) {
      turns.push({
        source: "speaker",
        speaker: currentSpeaker,
        text: currentLines.join(" "),
      });
    }
    currentSpeaker = null;
    currentLines = [];
  }

  for (const p of content) {
    const match = TIMESTAMP_SPEAKER_RE.exec(p.text);
    if (match) {
      flushTurn();
      currentSpeaker = match[2]!;
    } else if (p.text.trim() === "") {
      // Empty line — don't flush yet (Google transcripts have blanks between turns)
    } else if (currentSpeaker) {
      currentLines.push(p.text.trim());
    }
  }
  flushTurn();

  return turns;
}

/** Parse a Google Meet notes doc into summary text */
export function parseMeetingNotesDoc(content: DocParagraph[]): string {
  return content
    .map((p) => p.text)
    .filter((t) => t.length > 0)
    .join("\n");
}
```

```typescript
import type { MeetingNoteProvider, ProviderMeetingData } from "./types.js";

export class GoogleMeetProvider implements MeetingNoteProvider {

  async isAvailable(userId: string): Promise<boolean> {
    const account = await prisma.googleAccount.findFirst({
      where: { userId, hasDriveScope: true },
      select: { id: true },
    });
    return !!account;

  async fetchForEvent(userId: string, calendarEvent: CalendarEvent): Promise<ProviderMeetingData | null> {
    const account = await prisma.googleAccount.findFirst({
      where: { userId, hasDriveScope: true },
    });
    if (!account) return null;

    // Only process events with a meeting link (Google Meet, Zoom, etc.)
    if (!calendarEvent.meetingLink?.includes("meet.google.com")) return null;

    const driveClient = getDriveClient(account);
    const docsClient = getDocsClient(account);

    const attachments = calendarEvent.attachments as Array<{ fileId?: string; title?: string; mimeType?: string }> | null;

    const { transcriptFileId, notesFileId } = await findMeetArtifacts(
      driveClient,
      attachments,
      calendarEvent.title,
      calendarEvent.startTime,
      calendarEvent.endTime,
    );

    if (!transcriptFileId && !notesFileId) return null;

    let transcript = null;
    let summary = null;
    const rawData: Record<string, unknown> = {};

    if (transcriptFileId) {
      try {
        const content = await readDocContent(docsClient, transcriptFileId);
        transcript = parseTranscriptDoc(content);
        rawData.transcriptFileId = transcriptFileId;
      } catch (err) {
        console.warn(`[google-meet] Failed to parse transcript ${transcriptFileId}:`, err);
      }
    }

    if (notesFileId) {
      try {
        const content = await readDocContent(docsClient, notesFileId);
        summary = parseMeetingNotesDoc(content);
        rawData.notesFileId = notesFileId;
      } catch (err) {
        console.warn(`[google-meet] Failed to parse notes ${notesFileId}:`, err);
      }
    }

    if (!transcript && !summary) return null;

    const externalId = transcriptFileId ?? notesFileId!;

    return {
      provider: "google_meet",
      externalId,
      accountId: account.id,
      calendarEventId: calendarEvent.id,
      title: calendarEvent.title,
      summary,
      transcript,
      attendees: (calendarEvent.attendees as MeetingNoteAttendee[] | null) ?? null,
      meetingStartedAt: calendarEvent.startTime,
      meetingEndedAt: calendarEvent.endTime,
      rawData,
    };
  }

  async fetchRecent(userId: string, since: Date, until: Date): Promise<ProviderMeetingData[]> {
    const account = await prisma.googleAccount.findFirst({
      where: { userId, hasDriveScope: true },
    });
    if (!account) return [];

    // Find calendar events with Google Meet links in the time range
    const events = await prisma.calendarEvent.findMany({
      where: {
        userId,
        startTime: { gte: since },
        endTime: { lte: until },
        meetingLink: { contains: "meet.google.com" },
        isAllDay: false,
      },
    });

    const results: ProviderMeetingData[] = [];
    for (const event of events) {
      try {
        const data = await this.fetchForEvent(userId, event);
        if (data) results.push(data);
      } catch (err) {
        console.warn(`[google-meet] Failed to fetch for event ${event.id}:`, err);
      }
    }

    return results;
  }
}
```

```typescript
import type { CalendarEvent, Prisma } from "@prisma/client";
import type { MeetingProviderRegistry } from "./registry.js";
import type { ProviderMeetingData } from "./types.js";
import { mergeMeetingNoteFields, type MergeInput } from "./merge.js";
import { scrubProviderRawData } from "./scrub.js";
import { prisma } from "../../lib/prisma.js";
import { publishSSE } from "../../lib/sse.js";
import { enqueueEmbed } from "@brett/ai";
import { processActionItems } from "../granola-action-items.js";
import type { MeetingTranscriptTurn, MeetingNoteAttendee } from "@brett/types";

export class MeetingNoteCoordinator {
  constructor(private registry: MeetingProviderRegistry) {}

  /**
   * Sync meeting notes for a specific calendar event.
   * Called by the post-meeting cron (every 5 minutes).
   */
  async syncForEvent(userId: string, calendarEvent: CalendarEvent): Promise<void> {
    const providers = await this.registry.getAvailable(userId);
    if (providers.length === 0) return;

    // Fetch from all providers in parallel
    const fetchResults = await Promise.allSettled(
      providers.map((p) => p.fetchForEvent(userId, calendarEvent)),
    );

    const providerData: ProviderMeetingData[] = [];
    for (const result of fetchResults) {
      if (result.status === "fulfilled" && result.value) {
        providerData.push(result.value);
      } else if (result.status === "rejected") {
        console.error("[coordinator] Provider fetch failed:", result.reason);
      }
    }

    if (providerData.length === 0) return;

    // Merge sequentially to prevent duplicate MeetingNote race
    for (const data of providerData) {
      await this.mergeProviderData(userId, calendarEvent.id, data);
    }
  }

  /**
   * Sync meeting notes for a time range.
   * Called by the periodic sweep cron (every 30 minutes).
   */
  async syncRecent(userId: string, since: Date, until: Date): Promise<void> {
    const providers = await this.registry.getAvailable(userId);
    if (providers.length === 0) return;

    for (const provider of providers) {
      try {
        const results = await provider.fetchRecent(userId, since, until);
        for (const data of results) {
          // Try to find the calendar event for this data
          const calendarEventId = data.calendarEventId ?? await this.findCalendarEventId(userId, data);
          if (!calendarEventId) continue; // Can't merge without a calendar event

          await this.mergeProviderData(userId, calendarEventId, data);
        }
      } catch (err) {
        console.error(`[coordinator] ${provider.provider} fetchRecent failed:`, err);
      }
    }
  }

  /**
   * Initial backfill for a provider (e.g., on first connection or re-auth).
   */
  async initialSync(userId: string, providerName: string): Promise<number> {
    const providers = await this.registry.getAvailable(userId);
    const provider = providers.find((p) => p.provider === providerName);
    if (!provider) return 0;

    const since = new Date(Date.now() - 30 * 24 * 60 * 60 * 1000); // 30 days ago
    const until = new Date();

    const results = await provider.fetchRecent(userId, since, until);
    let count = 0;

    for (const data of results) {
      const calendarEventId = data.calendarEventId ?? await this.findCalendarEventId(userId, data);
      if (!calendarEventId) continue;

      try {
        await this.mergeProviderData(userId, calendarEventId, data);
        count++;
      } catch (err) {
        console.error(`[coordinator] Initial sync merge failed:`, err);
      }
    }

    return count;
  }

  private async mergeProviderData(
    userId: string,
    calendarEventId: string,
    data: ProviderMeetingData,
  ): Promise<void> {
    // Check if this exact source already exists
    const existingSource = await prisma.meetingNoteSource.findUnique({
      where: { provider_externalId: { provider: data.provider, externalId: data.externalId } },
    });
    if (existingSource) return; // Already processed

    // Scrub raw data before storage
    const scrubbedRawData = scrubProviderRawData(data.provider, data.rawData);

    // Upsert MeetingNote — one per (userId, calendarEventId)
    const existingNote = await prisma.meetingNote.findUnique({
      where: { userId_calendarEventId: { userId, calendarEventId } },
    });

    const existing: MergeInput = existingNote
      ? {
          title: existingNote.title,
          summary: existingNote.summary,
          transcript: existingNote.transcript as MeetingTranscriptTurn[] | null,
          attendees: existingNote.attendees as MeetingNoteAttendee[] | null,
          sources: existingNote.sources,
        }
      : { title: null, summary: null, transcript: null, attendees: null, sources: [] };

    const merged = mergeMeetingNoteFields(existing, {
      provider: data.provider,
      title: data.title,
      summary: data.summary,
      transcript: data.transcript,
      attendees: data.attendees,
    });

    const isFirstSource = !existingNote;

    // Build account FK fields based on provider
    const accountFks: Record<string, string | null> = {
      granolaAccountId: data.provider === "granola" ? data.accountId : null,
      googleAccountId: data.provider === "google_meet" ? data.accountId : null,
    };

    if (isFirstSource) {
      // Create new MeetingNote + MeetingNoteSource in transaction
      const note = await prisma.$transaction(async (tx) => {
        const created = await tx.meetingNote.create({
          data: {
            userId,
            calendarEventId,
            provider: data.provider,
            title: merged.title,
            summary: merged.summary,
            transcript: merged.transcript ?? Prisma.DbNull,
            attendees: merged.attendees ?? Prisma.DbNull,
            sources: merged.sources,
            meetingStartedAt: data.meetingStartedAt,
            meetingEndedAt: data.meetingEndedAt,
          },
        });

        await tx.meetingNoteSource.create({
          data: {
            meetingNoteId: created.id,
            userId,
            provider: data.provider,
            externalId: data.externalId,
            ...accountFks,
            title: data.title,
            summary: data.summary,
            transcript: data.transcript ?? Prisma.DbNull,
            attendees: data.attendees ?? Prisma.DbNull,
            rawData: scrubbedRawData ?? Prisma.DbNull,
          },
        });

        return created;
      });

      // Embed and extract action items for first source
      enqueueEmbed({ entityType: "meeting_note", entityId: note.id, userId });

      if (merged.summary) {
        processActionItems(
          note.id,
          calendarEventId,
          userId,
          merged.summary,
          merged.title,
          data.meetingStartedAt,
          (merged.attendees ?? []) as { name: string; email: string }[],
        ).catch((err) => console.error("[coordinator] Action item extraction failed:", err));
      }
    } else {
      // Merge into existing MeetingNote
      await prisma.$transaction(async (tx) => {
        await tx.meetingNote.update({
          where: { id: existingNote.id },
          data: {
            title: merged.title,
            summary: merged.summary,
            transcript: merged.transcript ?? Prisma.DbNull,
            attendees: merged.attendees ?? Prisma.DbNull,
            sources: merged.sources,
          },
        });

        await tx.meetingNoteSource.create({
          data: {
            meetingNoteId: existingNote.id,
            userId,
            provider: data.provider,
            externalId: data.externalId,
            ...accountFks,
            title: data.title,
            summary: data.summary,
            transcript: data.transcript ?? Prisma.DbNull,
            attendees: data.attendees ?? Prisma.DbNull,
            rawData: scrubbedRawData ?? Prisma.DbNull,
          },
        });
      });

      // Re-embed (don't re-extract action items)
      enqueueEmbed({ entityType: "meeting_note", entityId: existingNote.id, userId });
    }

    // SSE notification
    publishSSE(userId, {
      type: "meeting.note.synced",
      payload: { calendarEventId, sources: merged.sources },
    });
  }

  /**
   * Find a calendar event ID for provider data that doesn't have a direct link.
   * Uses the existing meeting-matcher fuzzy matching.
   */
  private async findCalendarEventId(userId: string, data: ProviderMeetingData): Promise<string | null> {
    const { findBestMatch } = await import("../meeting-matcher.js");
    const windowStart = new Date(data.meetingStartedAt.getTime() - 3 * 60 * 60 * 1000);
    const windowEnd = new Date(data.meetingEndedAt.getTime() + 3 * 60 * 60 * 1000);

    const candidates = await prisma.calendarEvent.findMany({
      where: {
        userId,
        startTime: { gte: windowStart },
        endTime: { lte: windowEnd },
        isAllDay: false,
      },
      select: { id: true, title: true, startTime: true, endTime: true, attendees: true },
    });

    const match = findBestMatch(
      {
        title: data.title,
        startTime: data.meetingStartedAt,
        endTime: data.meetingEndedAt,
        attendees: data.attendees ?? [],
      },
      candidates.map((e) => ({
        id: e.id,
        title: e.title,
        startTime: e.startTime,
        endTime: e.endTime,
        attendees: (e.attendees as Array<{ email: string }>) ?? [],
      })),
    );

    return match?.id ?? null;
  }
}
```
